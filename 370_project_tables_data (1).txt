CREATE TABLE flights (
  flight_id VARCHAR(30) PRIMARY KEY,
  source VARCHAR(30),
  destination VARCHAR(30),
  date DATE,
  boarding_time TIME,
  airline VARCHAR(30),
  pilot_id VARCHAR(30),
  hangar_id VARCHAR(30),
  capacity INT,
  booked INT,
  FOREIGN KEY (pilot_id) REFERENCES pilots(pilot_id),
  FOREIGN KEY (hangar_id) REFERENCES hangar(hangar_id),
  FOREIGN KEY (airline) REFERENCES airlines(airline_name)
);


CREATE TABLE tickets (
  ticket_number INT PRIMARY KEY,
  flight_id VARCHAR(30),
  price DECIMAL(7, 5),
  seat_number VARCHAR(30),
  class VARCHAR(30),
  FOREIGN KEY (flight_id) REFERENCES flights(flight_id)
);


CREATE TABLE hangar (
  hangar_id VARCHAR(30) PRIMARY KEY,
  hangar_capacity INT
);


CREATE TABLE users (
  first_name VARCHAR(30),
  last_name VARCHAR(30),
  username VARCHAR(30) PRIMARY KEY,
  passport VARCHAR(30),
  nid VARCHAR(30),
  date_of_birth DATE,
  city VARCHAR(30),
  country VARCHAR(30),
  phone_no int,
  email VARCHAR(50),
  password VARCHAR(50)
);


CREATE TABLE admins (
  first_name VARCHAR(30),
  last_name VARCHAR(30),
  admin_id VARCHAR(30) PRIMARY KEY,
  password VARCHAR(50)
);

CREATE TABLE registration_code (
  registration_code VARCHAR(30) PRIMARY KEY,
);


CREATE TABLE booked_flights (
  username VARCHAR(30),
  flight_id VARCHAR(30),
  ticket_number INT PRIMARY KEY,
  price DECIMAL(7, 5),
  seat_number VARCHAR(30),
  class VARCHAR(30),
  FOREIGN KEY (username) REFERENCES users(username),
  FOREIGN KEY (flight_id) REFERENCES flights(flight_id)
);


CREATE TABLE other_staff (
  first_name VARCHAR(30),
  last_name VARCHAR(30),
  id VARCHAR(30) PRIMARY KEY,
  job VARCHAR(30),
  hangar_id VARCHAR(30),
  FOREIGN KEY (hangar_id) REFERENCES hangar(hangar_id)
);


CREATE TABLE airlines (
  airline_name VARCHAR(30) PRIMARY KEY,
  no_of_planes INT,
  no_of_pilots INT
);


CREATE TABLE pilots (
  first_name VARCHAR(30),
  last_name VARCHAR(30),
  pilot_id VARCHAR(30) PRIMARY KEY,
  airline_name VARCHAR(30)
  FOREIGN KEY (airline_name) REFERENCES airlines(airline_name)
);
